660 virtual QCString trWriteList(int numEntries)
661 {
662 QCString result;
663 int i;
664 // the inherits list contain `numEntries' classes
665 for (i=0;i<numEntries;i++)
666 {
667 // use generateMarker to generate placeholders for the class links!
668 result+=generateMarker(i); // generate marker for entry i in the list
669 // (order is left to right)
670
671 if (i!=numEntries-1) // not the last entry, so we need a separator
672 {
673 if (i<numEntries-2) // not the fore last entry
674 result+=", ";
675 else // the fore last entry
676 result+=", og ";
677 }
678 }
679 return result;
680 }
