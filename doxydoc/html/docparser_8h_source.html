155template<class T>
156class CompAccept : public DocNode
157{
158 public:
159 CompAccept(DocParser &parser) : DocNode(parser) {}
160 void accept(DocVisitor *v) override
161 {
162 T *obj = dynamic_cast<T *>(this);
163 v->visitPre(obj);
164 for (const auto &n : m_children) n->accept(v);
165 v->visitPost(obj);
166 }
167 const DocNodeList &children() const { return m_children; }
168 DocNodeList &children() { return m_children; }
169
170 protected:
171 DocNodeList m_children;
172};
